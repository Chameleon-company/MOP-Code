# Function to collect data
def collect_data(dataset_id):
    base_url = 'https://data.melbourne.vic.gov.au/api/explore/v2.1/catalog/datasets/'
    format = 'csv'
    url = f'{base_url}{dataset_id}/exports/{format}'
    params = {
        'select': '*',
        'limit': -1,  # retrieve all records
        'lang': 'en',
        'timezone': 'UTC',
    }

    # GET request to the dataset API
    response = requests.get(url, params=params)

    if response.status_code == 200:
        # Decoding the content and reading into a pandas dataframe
        url_content = response.content.decode('utf-8')
        # Detecting delimiter
        delimiter = ','
        sample_lines = url_content.split('\n')[:5]
        for line in sample_lines:
            if ';' in line:
                delimiter = ';'
                break
        # Reading data into a pandas DataFrame
        df = pd.read_csv(StringIO(url_content), delimiter=delimiter)
        print(f'Successfully retrieved dataset: {dataset_id}')
        return df
    else:
        print(f'Request failed with status code {response.status_code}')
        return None

A skewness value of 13.94 indicates that the distribution of Total jobs is highly positively skewed, meaning there are a few businesses with a very large number of jobs, while most businesses have fewer jobs.

Given this high skewness, its likely better to impute the missing values using the median, which is less affected by outliers compared to the mean.
