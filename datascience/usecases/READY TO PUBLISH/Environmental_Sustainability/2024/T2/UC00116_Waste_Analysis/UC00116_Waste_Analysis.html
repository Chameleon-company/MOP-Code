# Function to collect data
def collect_data(dataset_id):
    base_url = 'https://data.melbourne.vic.gov.au/api/explore/v2.1/catalog/datasets/'
    format = 'csv'
    url = f'{base_url}{dataset_id}/exports/{format}'
    params = {
        'select': '*',
        'limit': -1,  # all records
        'lang': 'en',
        'timezone': 'UTC',
    }

    # GET request
    response = requests.get(url, params=params)

    if response.status_code == 200:
        # StringIO to read the CSV data
        url_content = response.content.decode('utf-8')
        # Try to determine the delimiter by reading the first few lines
        delimiter = ','
        sample_lines = url_content.split('\n')[:5]
        for line in sample_lines:
            if ';' in line:
                delimiter = ';'
                break
        df = pd.read_csv(StringIO(url_content), delimiter=delimiter)
        print(f'Successfully retrieved dataset: {dataset_id}')
        return df
    else:
        print(f'Request failed with status code {response.status_code}')
        return None

def engineer_features(df):
    # Add day of week
    df['day_of_week'] = df['date'].dt.dayofweek
    
    # Add month
    df['month'] = df['date'].dt.month
    
    # Add season
    df['season'] = df['date'].dt.month.map({1: 'Summer', 2: 'Summer', 3: 'Autumn', 
                                            4: 'Autumn', 5: 'Autumn', 6: 'Winter',
                                            7: 'Winter', 8: 'Winter', 9: 'Spring', 
                                            10: 'Spring', 11: 'Spring', 12: 'Summer'})
    
    # Add is_weekend flag
    df['is_weekend'] = df['day_of_week'].isin([5, 6]).astype(int)
    
    # Add rolling average (7-day window)
    df['rolling_avg_7d'] = df['daily_estimated_net_weight_kg'].rolling(window=7).mean()
    
    # Fill NaN values in rolling_avg_7d with the overall mean
    overall_mean = df['daily_estimated_net_weight_kg'].mean()
    df['rolling_avg_7d'].fillna(overall_mean, inplace=True)
    
    return df

def analyze_event_correlation(waste_df, events_df):
    # Convert event_start to datetime
    events_df['event_start'] = pd.to_datetime(events_df['event_start'])
    
    # Count events per day
    event_counts = events_df.groupby(events_df['event_start'].dt.date).size().reset_index(name='event_count')
    event_counts['event_start'] = pd.to_datetime(event_counts['event_start'])
    
    # Merge waste data with event counts
    merged_df = pd.merge(waste_df, event_counts, left_on='date', right_on='event_start', how='left')
    merged_df['event_count'].fillna(0, inplace=True)
    
    # Calculate correlation
    correlation = merged_df['daily_estimated_net_weight_kg'].corr(merged_df['event_count'])
    
    # Visualize relationship
